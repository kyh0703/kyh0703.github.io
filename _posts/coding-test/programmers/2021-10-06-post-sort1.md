```c++
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

vector<int> solution(vector<int> array, vector<vector<int>> commands) {
    vector<int> answer;
    for (auto cmd : commands)
    {
        int i = cmd[0] - 1, j = cmd[1], k = cmd[2] - 1;
        vector<int> copy_array;
        copy_array.resize(j - i);
        copy(array.begin() + i, array.begin() + (i + j), copy_array.begin());
        sort(copy_array.begin(), copy_array.end());
        answer.push_back(copy_array[k]);
    }
    
    return answer;
}
```
